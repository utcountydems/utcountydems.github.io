ballot_section:
  tag: "Procedures"
  heading: "Ballot & Voting Procedures"
  sub: "Follow the Democratic Party's official voting procedures to ensure every election is fair and valid."
  cards:
    - icon: "fa-solid fa-scroll"
      title: "Paper Ballots"
      desc: "All elections use <strong>written paper ballots</strong>. Pre-cut blank paper into uniform slips before the event. Voters write the name of their chosen candidate on the slip and hand it to a designated teller. No pre-printed ballots with names are used."
    - icon: "fa-solid fa-user-check"
      title: "Tellers"
      desc: "Designate at least two tellers after everyone splits into precincts. Tellers collect ballots, count them, and report totals to the Chair. Tellers should not be candidates in the race being counted. Results are announced publicly."
    - icon: "fa-solid fa-scale-balanced"
      title: "Majority Rule"
      desc: "A candidate wins with a <strong>simple majority</strong> of votes cast for that position. If no candidate achieves a majority in the first round (multiple delegate seats), additional rounds of voting are conducted until all seats are filled."
  callout:
    heading: "Official Caucus Rules"
    body: "For the complete official procedures, see the <a href=\"/caucus-rules.html\">Caucus Rules page</a> and the Utah Democratic Party bylaws. If a procedural dispute arises during your caucus, contact the county party immediately at <a href=\"mailto:communications@utcountydems.com\">communications@utcountydems.com</a>."
